var store={txs:[],inv:[],user:null};
var userType='';

document.addEventListener('DOMContentLoaded',function(){
// --- Persistence helpers ---
function saveStore(){
	try{
		localStorage.setItem('raj_store',JSON.stringify(store));
	}catch(e){console.warn('Could not save to localStorage',e)}
}
function loadStore(){
	try{
		var data=localStorage.getItem('raj_store');
		if(data){
			var parsed=JSON.parse(data);
			store.txs=parsed.txs||[];
			store.inv=parsed.inv||[];
			store.user=parsed.user||null;
		}
	}catch(e){console.warn('Could not load from localStorage',e)}
}

// initialize from storage
loadStore();

function updateInv(){
var h='';
store.inv.forEach(function(i){
h+='<tr><td>'+i.name+'</td><td>'+i.stock+'</td><td>$'+i.cost+'</td></tr>';
});
document.getElementById('invTable').innerHTML=h||'<tr><td colspan="3" class="text-center">No items</td></tr>';
}

function updateStats(){
var d=0,m=0,y=0;
store.txs.forEach(function(t){
d+=t.amt;
m+=t.amt;
y+=t.amt;
});
document.getElementById('repDay').textContent='$'+d.toFixed(2);
document.getElementById('repMonth').textContent='$'+m.toFixed(2);
document.getElementById('repYear').textContent='$'+y.toFixed(2);
var h='';
store.txs.slice(-10).reverse().forEach(function(t){
